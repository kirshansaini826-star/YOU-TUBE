import { useState, useRef, useEffect } from "react";

const MODES = [
  { id: "chat", label: "💬 Chat", desc: "Ask me anything" },
  { id: "write", label: "✍️ Writing", desc: "Help with writing tasks" },
  { id: "image", label: "🖼️ Image", desc: "Analyze images" },
  { id: "quiz", label: "🧠 Quiz", desc: "Generate quizzes" },
];

const WRITE_TYPES = ["Improve my text", "Summarize", "Translate", "Make it formal", "Make it casual", "Expand", "Shorten"];
const QUIZ_TOPICS = ["Science", "History", "Math", "Technology", "General Knowledge", "Custom topic..."];

function Spinner() {
  return (
    <div style={{ display: "flex", gap: 5, alignItems: "center", padding: "10px 0" }}>
      {[0, 1, 2].map(i => (
        <div key={i} style={{
          width: 8, height: 8, borderRadius: "50%", background: "#7c5cfc",
          animation: "bounce 1.2s infinite", animationDelay: `${i * 0.2}s`
        }} />
      ))}
    </div>
  );
}

async function callClaude(messages, systemPrompt) {
  const response = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-6",
      max_tokens: 1000,
      system: systemPrompt || "You are a helpful AI assistant.",
      messages,
    }),
  });
  const data = await response.json();
  return data.content?.map(b => b.text || "").join("") || "No response.";
}

// ── CHAT MODE ──
function ChatMode() {
  const [messages, setMessages] = useState([
    { role: "assistant", content: "Hi! I'm your AI assistant. Ask me anything — I'm here to help! 🌟" }
  ]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const bottomRef = useRef();

  useEffect(() => { bottomRef.current?.scrollIntoView({ behavior: "smooth" }); }, [messages, loading]);

  async function send() {
    if (!input.trim() || loading) return;
    const userMsg = { role: "user", content: input };
    setMessages(prev => [...prev, userMsg]);
    setInput("");
    setLoading(true);
    const history = [...messages, userMsg].filter(m => m.role !== "assistant" || messages.indexOf(m) > 0)
      .map(m => ({ role: m.role, content: m.content }));
    const reply = await callClaude(history, "You are a friendly, knowledgeable AI assistant. Be concise but thorough.");
    setMessages(prev => [...prev, { role: "assistant", content: reply }]);
    setLoading(false);
  }

  return (
    <div style={{ display: "flex", flexDirection: "column", height: "100%" }}>
      <div style={{ flex: 1, overflowY: "auto", padding: "16px", display: "flex", flexDirection: "column", gap: 12 }}>
        {messages.map((m, i) => (
          <div key={i} style={{ display: "flex", justifyContent: m.role === "user" ? "flex-end" : "flex-start" }}>
            <div style={{
              maxWidth: "80%", padding: "10px 14px", borderRadius: m.role === "user" ? "18px 18px 4px 18px" : "18px 18px 18px 4px",
              background: m.role === "user" ? "linear-gradient(135deg,#7c5cfc,#a78bfa)" : "#1e1e2e",
              color: "#e2e0ff", fontSize: 14, lineHeight: 1.6, whiteSpace: "pre-wrap"
            }}>{m.content}</div>
          </div>
        ))}
        {loading && <div style={{ display: "flex" }}><div style={{ background: "#1e1e2e", borderRadius: "18px 18px 18px 4px", padding: "8px 14px" }}><Spinner /></div></div>}
        <div ref={bottomRef} />
      </div>
      <div style={{ padding: "12px 16px", borderTop: "1px solid #2a2a3e", display: "flex", gap: 8 }}>
        <input
          value={input} onChange={e => setInput(e.target.value)}
          onKeyDown={e => e.key === "Enter" && !e.shiftKey && send()}
          placeholder="Type a message…"
          style={{ flex: 1, background: "#1e1e2e", border: "1px solid #3a3a5e", borderRadius: 12, padding: "10px 14px", color: "#e2e0ff", fontSize: 14, outline: "none" }}
        />
        <button onClick={send} disabled={loading || !input.trim()}
          style={{ background: "linear-gradient(135deg,#7c5cfc,#a78bfa)", border: "none", borderRadius: 12, padding: "10px 18px", color: "#fff", fontWeight: 700, cursor: "pointer", opacity: loading || !input.trim() ? 0.5 : 1 }}>
          Send
        </button>
      </div>
    </div>
  );
}

// ── WRITING MODE ──
function WriteMode() {
  const [type, setType] = useState(WRITE_TYPES[0]);
  const [input, setInput] = useState("");
  const [output, setOutput] = useState("");
  const [loading, setLoading] = useState(false);

  async function run() {
    if (!input.trim() || loading) return;
    setLoading(true); setOutput("");
    const prompt = `Task: ${type}\n\nText:\n${input}`;
    const reply = await callClaude([{ role: "user", content: prompt }], "You are an expert writing assistant. Perform the requested writing task clearly and professionally.");
    setOutput(reply);
    setLoading(false);
  }

  return (
    <div style={{ padding: 16, display: "flex", flexDirection: "column", gap: 12, height: "100%", overflowY: "auto" }}>
      <div style={{ display: "flex", flexWrap: "wrap", gap: 8 }}>
        {WRITE_TYPES.map(t => (
          <button key={t} onClick={() => setType(t)} style={{
            padding: "6px 14px", borderRadius: 20, border: "1px solid #3a3a5e", cursor: "pointer", fontSize: 13,
            background: type === t ? "linear-gradient(135deg,#7c5cfc,#a78bfa)" : "#1e1e2e",
            color: type === t ? "#fff" : "#a0a0c0", fontWeight: type === t ? 700 : 400
          }}>{t}</button>
        ))}
      </div>
      <textarea value={input} onChange={e => setInput(e.target.value)} placeholder="Paste or type your text here…"
        style={{ background: "#1e1e2e", border: "1px solid #3a3a5e", borderRadius: 12, padding: 12, color: "#e2e0ff", fontSize: 14, minHeight: 120, resize: "vertical", outline: "none", lineHeight: 1.6 }} />
      <button onClick={run} disabled={loading || !input.trim()}
        style={{ background: "linear-gradient(135deg,#7c5cfc,#a78bfa)", border: "none", borderRadius: 12, padding: "12px", color: "#fff", fontWeight: 700, cursor: "pointer", opacity: loading || !input.trim() ? 0.5 : 1, fontSize: 15 }}>
        {loading ? "Working…" : `✨ ${type}`}
      </button>
      {loading && <Spinner />}
      {output && (
        <div style={{ background: "#1e1e2e", border: "1px solid #3a3a5e", borderRadius: 12, padding: 14, color: "#e2e0ff", fontSize: 14, whiteSpace: "pre-wrap", lineHeight: 1.7 }}>
          <div style={{ color: "#7c5cfc", fontWeight: 700, marginBottom: 8, fontSize: 12, textTransform: "uppercase", letterSpacing: 1 }}>Result</div>
          {output}
        </div>
      )}
    </div>
  );
}

// ── IMAGE MODE ──
function ImageMode() {
  const [image, setImage] = useState(null);
  const [preview, setPreview] = useState(null);
  const [question, setQuestion] = useState("");
  const [output, setOutput] = useState("");
  const [loading, setLoading] = useState(false);

  function handleFile(e) {
    const file = e.target.files[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = ev => { setPreview(ev.target.result); setImage({ data: ev.target.result.split(",")[1], type: file.type }); setOutput(""); };
    reader.readAsDataURL(file);
  }

  async function analyze() {
    if (!image || loading) return;
    setLoading(true); setOutput("");
    const response = await fetch("https://api.anthropic.com/v1/messages", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        model: "claude-sonnet-4-6", max_tokens: 1000,
        messages: [{ role: "user", content: [
          { type: "image", source: { type: "base64", media_type: image.type, data: image.data } },
          { type: "text", text: question || "Describe this image in detail." }
        ]}],
      }),
    });
    const data = await response.json();
    setOutput(data.content?.map(b => b.text || "").join("") || "Could not analyze image.");
    setLoading(false);
  }

  return (
    <div style={{ padding: 16, display: "flex", flexDirection: "column", gap: 12, height: "100%", overflowY: "auto" }}>
      <label style={{ border: "2px dashed #3a3a5e", borderRadius: 16, padding: 24, textAlign: "center", cursor: "pointer", color: "#7c5cfc", display: "block" }}>
        {preview ? <img src={preview} alt="upload" style={{ maxWidth: "100%", maxHeight: 200, borderRadius: 10, objectFit: "contain" }} />
          : <div><div style={{ fontSize: 36 }}>🖼️</div><div style={{ marginTop: 8, fontSize: 14, color: "#a0a0c0" }}>Click to upload an image</div></div>}
        <input type="file" accept="image/*" onChange={handleFile} style={{ display: "none" }} />
      </label>
      <input value={question} onChange={e => setQuestion(e.target.value)} placeholder="Ask something about the image… (optional)"
        style={{ background: "#1e1e2e", border: "1px solid #3a3a5e", borderRadius: 12, padding: "10px 14px", color: "#e2e0ff", fontSize: 14, outline: "none" }} />
      <button onClick={analyze} disabled={!image || loading}
        style={{ background: "linear-gradient(135deg,#7c5cfc,#a78bfa)", border: "none", borderRadius: 12, padding: "12px", color: "#fff", fontWeight: 700, cursor: "pointer", opacity: !image || loading ? 0.5 : 1, fontSize: 15 }}>
        {loading ? "Analyzing…" : "🔍 Analyze Image"}
      </button>
      {loading && <Spinner />}
      {output && (
        <div style={{ background: "#1e1e2e", border: "1px solid #3a3a5e", borderRadius: 12, padding: 14, color: "#e2e0ff", fontSize: 14, whiteSpace: "pre-wrap", lineHeight: 1.7 }}>
          <div style={{ color: "#7c5cfc", fontWeight: 700, marginBottom: 8, fontSize: 12, textTransform: "uppercase", letterSpacing: 1 }}>Analysis</div>
          {output}
        </div>
      )}
    </div>
  );
}

// ── QUIZ MODE ──
function QuizMode() {
  const [topic, setTopic] = useState(QUIZ_TOPICS[0]);
  const [custom, setCustom] = useState("");
  const [quiz, setQuiz] = useState(null);
  const [answers, setAnswers] = useState({});
  const [submitted, setSubmitted] = useState(false);
  const [loading, setLoading] = useState(false);

  async function generate() {
    setLoading(true); setQuiz(null); setAnswers({}); setSubmitted(false);
    const t = topic === "Custom topic..." ? custom : topic;
    const prompt = `Generate a 5-question multiple choice quiz about "${t}". Return ONLY a JSON array like:
[{"q":"Question?","options":["A","B","C","D"],"answer":0}]
Where answer is the index (0-3) of the correct option. No markdown, no explanation.`;
    const reply = await callClaude([{ role: "user", content: prompt }], "You are a quiz generator. Return only valid JSON.");
    try {
      const clean = reply.replace(/```json|```/g, "").trim();
      setQuiz(JSON.parse(clean));
    } catch { setQuiz([{ q: "Couldn't parse quiz. Try again!", options: [], answer: 0 }]); }
    setLoading(false);
  }

  const score = quiz ? quiz.reduce((s, q, i) => s + (answers[i] === q.answer ? 1 : 0), 0) : 0;

  return (
    <div style={{ padding: 16, display: "flex", flexDirection: "column", gap: 12, height: "100%", overflowY: "auto" }}>
      <div style={{ display: "flex", flexWrap: "wrap", gap: 8 }}>
        {QUIZ_TOPICS.map(t => (
          <button key={t} onClick={() => setTopic(t)} style={{
            padding: "6px 14px", borderRadius: 20, border: "1px solid #3a3a5e", cursor: "pointer", fontSize: 13,
            background: topic === t ? "linear-gradient(135deg,#7c5cfc,#a78bfa)" : "#1e1e2e",
            color: topic === t ? "#fff" : "#a0a0c0", fontWeight: topic === t ? 700 : 400
          }}>{t}</button>
        ))}
      </div>
      {topic === "Custom topic..." && (
        <input value={custom} onChange={e => setCustom(e.target.value)} placeholder="Enter your topic…"
          style={{ background: "#1e1e2e", border: "1px solid #3a3a5e", borderRadius: 12, padding: "10px 14px", color: "#e2e0ff", fontSize: 14, outline: "none" }} />
      )}
      <button onClick={generate} disabled={loading || (topic === "Custom topic..." && !custom.trim())}
        style={{ background: "linear-gradient(135deg,#7c5cfc,#a78bfa)", border: "none", borderRadius: 12, padding: "12px", color: "#fff", fontWeight: 700, cursor: "pointer", opacity: loading ? 0.5 : 1, fontSize: 15 }}>
        {loading ? "Generating…" : "🧠 Generate Quiz"}
      </button>
      {loading && <Spinner />}
      {quiz && quiz.map((q, i) => (
        <div key={i} style={{ background: "#1e1e2e", border: "1px solid #3a3a5e", borderRadius: 12, padding: 14 }}>
          <div style={{ color: "#e2e0ff", fontWeight: 600, marginBottom: 10, fontSize: 14 }}>{i + 1}. {q.q}</div>
          {q.options.map((opt, j) => {
            let bg = "#2a2a3e";
            if (submitted) bg = j === q.answer ? "#1a3a2a" : answers[i] === j ? "#3a1a1a" : "#2a2a3e";
            else if (answers[i] === j) bg = "#2a2a4e";
            return (
              <div key={j} onClick={() => !submitted && setAnswers(a => ({ ...a, [i]: j }))}
                style={{ background: bg, border: `1px solid ${submitted && j === q.answer ? "#4caf50" : submitted && answers[i] === j ? "#f44336" : answers[i] === j ? "#7c5cfc" : "#3a3a5e"}`, borderRadius: 8, padding: "8px 12px", marginBottom: 6, cursor: submitted ? "default" : "pointer", color: "#e2e0ff", fontSize: 13, display: "flex", alignItems: "center", gap: 8 }}>
                <span style={{ width: 22, height: 22, borderRadius: "50%", background: answers[i] === j ? "#7c5cfc" : "#3a3a5e", display: "inline-flex", alignItems: "center", justifyContent: "center", fontSize: 11, color: "#fff", flexShrink: 0 }}>{["A","B","C","D"][j]}</span>
                {opt}
                {submitted && j === q.answer && " ✅"}
                {submitted && answers[i] === j && j !== q.answer && " ❌"}
              </div>
            );
          })}
        </div>
      ))}
      {quiz && !submitted && Object.keys(answers).length === quiz.length && (
        <button onClick={() => setSubmitted(true)}
          style={{ background: "linear-gradient(135deg,#22c55e,#16a34a)", border: "none", borderRadius: 12, padding: "12px", color: "#fff", fontWeight: 700, cursor: "pointer", fontSize: 15 }}>
          Submit Answers
        </button>
      )}
      {submitted && (
        <div style={{ background: "#1e1e2e", border: "1px solid #7c5cfc", borderRadius: 12, padding: 16, textAlign: "center" }}>
          <div style={{ fontSize: 28, fontWeight: 800, color: "#7c5cfc" }}>{score}/{quiz.length}</div>
          <div style={{ color: "#a0a0c0", marginTop: 4 }}>{score === quiz.length ? "Perfect! 🎉" : score >= quiz.length / 2 ? "Good job! 👍" : "Keep practicing! 💪"}</div>
        </div>
      )}
    </div>
  );
}
